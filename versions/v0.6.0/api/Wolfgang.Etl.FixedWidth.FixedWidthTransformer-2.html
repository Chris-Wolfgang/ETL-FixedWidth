Transforms a stream of TSource records into TDestination records in a single streaming pass (#14) — the middle stage of an extract → transform → load pipeline. Compose it between a FixedWidthExtractor<TRecord> reading one fixed-width layout and a FixedWidthLoader<TRecord> writing another to reformat a file (reorder, add/remove, or reformat fields).

public sealed class FixedWidthTransformer<TSource, TDestination> : TransformerBase<TSource, TDestination, FixedWidthReport>, ITransformWithProgressAndCancellationAsync<TSource, TDestination, FixedWidthReport>, ITransformWithCancellationAsync<TSource, TDestination>, ITransformWithProgressAsync<TSource, TDestination, FixedWidthReport>, ITransformAsync<TSource, TDestination>, IAsyncDisposable, IDisposable where TSource : notnull where TDestination : notnull

Examples

using var extractor = new FixedWidthExtractor<LegacyRecord>(sourceReader);
using var transformer = new FixedWidthTransformer<LegacyRecord, ModernRecord>(
    legacy => new ModernRecord { Id = legacy.OldId, Name = legacy.FullName });
using var loader = new FixedWidthLoader<ModernRecord>(destinationWriter);

var modern = transformer.TransformAsync(extractor.ExtractAsync(token), token);
await loader.LoadAsync(modern, token);
